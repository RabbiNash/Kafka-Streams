Create ratings stream

CREATE STREAM RATINGS
   (RATINGID BIGINT, USERID INT, STARS INT, ROUTEID INT,RATINGTIME BIGINT, CHANNEL VARCHAR, MESSAGE varchar)
   WITH (KAFKA_TOPIC='ratings', VALUE_FORMAT='AVRO');

   CREATE TABLE USERS
   (USERID VARCHAR PRIMARY KEY, REGISTERTIME BIGINT, GENDER VARCHAR, REGIONID VARCHAR)
   WITH (KAFKA_TOPIC='users', VALUE_FORMAT='AVRO');

   CREATE TABLE INVENTORY
   (ID BIGINT PRIMARY KEY, PRODUCTID BIGINT, QUANTITY BIGINT)
   WITH (KAFKA_TOPIC='invent', VALUE_FORMAT='AVRO');

   CREATE STREAM PAGEVIEWS
   (VIEWTIME BIGINT, USERID VARCHAR, PAGEID varchar)
   WITH (KAFKA_TOPIC='pageviews', VALUE_FORMAT='AVRO');

   CREATE STREAM PAGEVIEWS_FEMALE
   AS SELECT USERS.USERID AS USERID, PAGEID, REGIONID
   FROM PAGEVIEWS LEFT JOIN USERS ON PAGEVIEWS.USERID = USERS.USERID
   WHERE GENDER = 'FEMALE'
   EMIT CHANGES;
